class Solution:
    def findPeakElement(self, nums: List[int]) -> int:

        n=len(nums)
        if n==1:
            return 0
        elif n==2:
            return nums.index(max(nums[0],nums[1]))
        else:
            for i in range(n):
                if i == 0 and nums[i]>=nums[i+1]:
                    return i
                elif i ==n-1 and nums[i]>=nums[i-1]:
                    return i
                elif nums[i-1]<nums[i] and nums[i+1]<nums[i]:
                    return i
